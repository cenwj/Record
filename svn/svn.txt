#!/bin/sh
path=/usr/local/Cellar/nginx/1.2.7/html/test
svnuser=da
svnpasswd=sdfsdf

#库路径
REPOS="$1"
#版本号
REV="$2"
#改动的文件
changed=$(svnlook changed -r $REV $REPOS)
#echo "$changed" >> ~/log.txt

content=`echo $changed | awk '{print $2}'`
for i in $content
do
svn update $path$i --username $svnuser --password $svnpasswd
done
